以下 `eslint.config.mjs` 配置适用于 React + TypeScript 项目。

依赖项：

```bash
npm add -D @eslint/js globals typescript-eslint eslint-plugin-react-hooks
npm add -D eslint-config-prettier eslint-plugin-prettier
```

配置如下：

```js
import pluginJs from '@eslint/js'
import eslintPluginPrettierRecommended from 'eslint-plugin-prettier/recommended'
import reactHooks from 'eslint-plugin-react-hooks'
import globals from 'globals'
import tsEslint from 'typescript-eslint'

/** @type {import('eslint').Linter.Config[]} */
export default tsEslint.config(
  // ↓ 需要忽略的目录和文件
  { ignores: ['**/dist'] },

  pluginJs.configs.recommended,
  ...tsEslint.configs.recommended,

  // ↓ 项目配置
  {
    files: ['./{src,tests}/**/*.{ts,tsx}'],
    plugins: {
      'react-hooks': reactHooks,
    },
    languageOptions: {
      ecmaVersion: 2020,
      globals: globals.browser,
    },
    rules: {
      ...reactHooks.configs.recommended.rules,
      // 自己的忽略规则 ...
    },
  },

  eslintPluginPrettierRecommended
)
```

使用需知：

- 这个规则带有 Prettier 的格式化检测，以及 React Hooks 名称检测；
- 注意除了代码中用到的依赖之外，还需要安装 `eslint-config-prettier`。
